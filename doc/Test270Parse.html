def test_timing
  start = Time::now
  X12::TEST_REPEAT.times do
    @r = @@parser.parse('270', @@m)
    test_ST
    test_L2000A_NM1
    test_L2000C_NM1
    test_L2000A_HL
    test_absent
  end
  finish = Time::now
  puts sprintf("Parses per second, 270: %.2f, elapsed: %.1f", X12::TEST_REPEAT.to_f/(finish-start), finish-start)
end
